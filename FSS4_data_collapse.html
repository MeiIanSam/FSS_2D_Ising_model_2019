def scaling(Q,T,L,a,b):
    T_c=2.269
    t=(T-T_c)/T_c
    y=Q*L**(-a/b)
    x=t*L**(1/b)
    return x,y

def scalingC(Q,T,L,a,b):
    T_c=2.269
    t=(T-T_c)/T_c
    y=Q/(np.log(L))
    x=t*L**(1/b)
    return x,y

Ls=[4,8,16,32]
alpha=0
beta=-1/8
gamma=7/4
nu=1
Name=['M','chi']
critical_exp=[beta,gamma]

for i in range(2,4):
    fig=plt.figure()
    for L in Ls:
        df=pd.read_csv('data L = %.d .csv'%L)
        Ts=df.iloc[:,1]
        Qs=df.iloc[:,i]
        xlist=[]
        ylist=[]
        for j in range(len(Qs)):
            x,y=scaling(Qs[j],Ts[j],L,critical_exp[i-2],nu)
            xlist.append(x)
            ylist.append(y)
        plt.plot(xlist,ylist,'x',label='L=%d'%L)
    plt.title('{} data collapse'.format(Name[i-2]))
    plt.legend()
